1153  //! \return a pair containing eigenvalues in a column vector in the first element of the pair
1154  //! and the correponding eigenvectors in the second element

1158

1166  unsigned iterations = 0;
1167

1169  while (true) {

1171  size_t p, q;
1172  T largest = 0;
1173  for (size_t i = 0; i < A.mRows; i++) {
1174  for (size_t j = 0; j < A.mCols; j++) {
1175  // it can't be in the diagonal

1179  q = j;

1186  // in the diagonal of A

1188  //eigenvalues are returned in a column matrix for convenience
1189  return eigsort(A.diagonal(), V);
1190  }
1191 
1192  iterations++;
1193 
1194  // else, perform a Jacobi rotation using this angle phi as reference
1195  double phi = (A(q, q) - A(p, p)) / (2 * A(p, q));
1196  double sign = (phi > 0) - (phi < 0);
1197  double t = phi == 0 ? 1 : 1 / (phi + sign * sqrt(pow(phi, 2) + 1));
1198  double cos = 1 / (sqrt(1 + pow(t, 2)));

1202  Matrix U = identity(A.mRows);

1207  U(q, p) = -sin;
