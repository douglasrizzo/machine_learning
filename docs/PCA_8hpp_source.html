32  void fit() {
33  MatrixD XMinusMean = X.minusMean(); // standardize columns to have 0 mean
34  MatrixD covariances = XMinusMean.cov(); // get covariance matrix of the data
35 
36  // get the sum of variances, this'll be useful later
37  double sumVar = 0;
38  for (size_t i = 0; i < covariances.nRows(); i++) {
39  sumVar += covariances(i, i);
40  }
41 
42  pair<MatrixD, MatrixD> eig = covariances.eigen(); // eigenvalues and eigenvectors of cov matrix
43  eigenvalues = eig.first;
44  eigenvectors = eig.second;
45 
46  // calculate the percentage of variance that each eigenvalue "explains"
47  percentages = MatrixD(eigenvalues.nRows(), eigenvalues.nCols());
48  cumPercentages = MatrixD(eigenvalues.nRows(), eigenvalues.nCols());
49  for (int i = 0; i < eigenvalues.nRows(); i++) {
50  percentages(i, 0) = eigenvalues(i, 0) / sumVar;
51  cumPercentages(i, 0) = i == 0 ? percentages(i, 0) : percentages(i, 0) + cumPercentages(i - 1, 0);
52  }
53  }
