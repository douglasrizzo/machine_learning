33  void fit() {
34  MatrixD XMinusMean = X.minusMean(); // standardize columns to have 0 mean
35  MatrixD covariances = XMinusMean.cov(); // get covariance matrix of the data
36 
37  // get the sum of variances, this'll be useful later
38  double sumVar = 0;
39  for (size_t i = 0; i < covariances.nRows(); i++) {
40  sumVar += covariances(i, i);
41  }
42 
43  pair<MatrixD, MatrixD> eig = covariances.eigen(); // eigenvalues and eigenvectors of cov matrix
44  eigenvalues = eig.first;
45  eigenvectors = eig.second;
46 
47  // calculate the percentage of variance that each eigenvalue "explains"
48  percentages = MatrixD(eigenvalues.nRows(), eigenvalues.nCols());
49  cumPercentages = MatrixD(eigenvalues.nRows(), eigenvalues.nCols());
50  for (int i = 0; i < eigenvalues.nRows(); i++) {
51  percentages(i, 0) = eigenvalues(i, 0) / sumVar;
52  cumPercentages(i, 0) = i == 0 ? percentages(i, 0) : percentages(i, 0) + cumPercentages(i - 1, 0);
53  }
54  }
